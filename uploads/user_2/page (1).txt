CREATE TABLE `users` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `email` varchar(255) NOT NULL,
  `nume` varchar(128) NOT NULL,
  `prenume` varchar(128) NOT NULL,
  `password_hash` varchar(255) NOT NULL,
  `user_type` varchar(128) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `email` (`email`)
) 
CREATE TABLE `user_profile` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `email` varchar(255) NOT NULL,
  `name` varchar(128) NOT NULL,
  `address` varchar(255) NOT NULL,
  `profile_picture` longblob NOT NULL,
  `phone_number` varchar(20) NOT NULL,
  `joining_date` date NOT NULL,
  `user_type` varchar(128) NOT NULL,
  PRIMARY KEY (`id`)
)
CREATE TABLE `tags` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `tag_name` varchar(128) NOT NULL,
  PRIMARY KEY (`id`)
)
CREATE TABLE `skills` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `skill_name` varchar(128) NOT NULL,
  PRIMARY KEY (`id`)
)
CREATE TABLE `reviews` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `reviewer_id` int(11) NOT NULL,
  `profile_id` int(11) NOT NULL,
  `review` varchar(255) DEFAULT NULL,
  `created_at` timestamp NOT NULL DEFAULT current_timestamp(),
  PRIMARY KEY (`id`)
) 
CREATE TABLE `project_tags` (
  `project_id` int(11) NOT NULL,
  `tag_id` int(11) NOT NULL,
  PRIMARY KEY (`project_id`, `tag_id`)
)
CREATE TABLE `project` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `owner_id` int(11) NOT NULL,
  `title` varchar(128) NOT NULL,
  `description` varchar(1024) NOT NULL,
  `project_type` varchar(255) NOT NULL,
  PRIMARY KEY (`id`)
)
CREATE TABLE `profile_tags` (
  `user_profile_id` int(11) NOT NULL,
  `tag_id` int(11) NOT NULL,
  PRIMARY KEY (`user_profile_id`, `tag_id`)
)
CREATE TABLE `portfolio_skills` (
  `portfolio_item_id` int(11) NOT NULL,
  `skill_id` int(11) NOT NULL,
  PRIMARY KEY (`portfolio_item_id`, `skill_id`)
)
CREATE TABLE `portfolio_item` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `owner_id` int(11) NOT NULL,
  `title` varchar(128) NOT NULL,
  `description` varchar(1024) NOT NULL,
  PRIMARY KEY (`id`)
) 
CREATE TABLE `images_projects` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `project_item_id` int(11) NOT NULL,
  `image_data` longblob NOT NULL,
  PRIMARY KEY (`id`)
) 
CREATE TABLE `images_portfolios` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `portfolio_item_id` int(11) NOT NULL,
  `image_data` longblob NOT NULL,
  PRIMARY KEY (`id`)
)
